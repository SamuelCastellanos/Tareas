library(ggplot2)
students <- data.frame(
  id = 1:20,
  hours_study = c(2,3,4,5,6,3,4,7,8,5,6,7,3,2,9,10,4,6,8,5) ,
  score = c(50,55,65,70,75,60,66,80,85,72,74,78,58,52,90,95,67,76,88,73) , 
  gender = rep(c("Male", "Female"), 10) , 
  major= rep(c("Engineering","Business","Arts","Science"), 5)
)
